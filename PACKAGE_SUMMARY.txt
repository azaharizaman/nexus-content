📖 EXAMPLES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// Create article
$article = $articleManager->createArticle(
    articleId: 'art-001',
    title: 'Getting Started',
    slug: 'getting-started',
    category: $category,
    textContent: '# Welcome...',
    authorId: 'user-123',
    isPublic: true
);

// Publish
$articleManager->publish('art-001');

// Update (creates new version)
$articleManager->updateContent('art-001', 'Updated content', 'user-456');

// Search
$results = $articleManager->search(SearchCriteria::forPublic('query'));
